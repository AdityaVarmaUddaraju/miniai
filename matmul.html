def matmul(m1,m2):
    (ar,ac),(br,bc) = m1.shape,m2.shape
    c = torch.zeros((ar,bc))
    for i in range(ar):
        for j in range(bc):
            for k in range(ac):
                c[i,j] += m1[i,k] * m2[k,j]
    return c
